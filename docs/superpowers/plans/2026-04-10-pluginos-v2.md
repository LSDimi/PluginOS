# PluginOS v2: Operations, Multi-File & Bootloader

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Expand PluginOS from 13 to 28 operations covering the 80/20 of designer workflows, add multi-file support so agents can work across open Figma files, and implement a bootloader pattern so the plugin auto-updates without reimporting.

**Architecture:** New operations follow the existing registry pattern (manifest + execute in dedicated files, imported by index.ts). Multi-file support upgrades the WebSocket server to track multiple clients keyed by fileKey, with an "active file" concept. The bootloader replaces the static webpack UI with a minimal HTML shell that fetches fresh code from the MCP server's HTTP endpoint at startup.

**Tech Stack:** TypeScript, Figma Plugin API, WebSocket (ws), MCP SDK, Vitest, webpack

```typescript
// packages/bridge-plugin/src/operations/write.ts
import { registerOperation } from "./registry";

// --- create_frame ---
registerOperation({
  manifest: {
    name: "create_frame",
    description:
      "Create a new frame on the current page with specified dimensions, position, and optional auto-layout.",
    category: "components",
    params: {
      name: { type: "string", required: true, description: "Frame name" },
      width: { type: "number", required: false, description: "Width in px (default: 100)" },
      height: { type: "number", required: false, description: "Height in px (default: 100)" },
      x: { type: "number", required: false, description: "X position (default: 0)" },
      y: { type: "number", required: false, description: "Y position (default: 0)" },
      auto_layout: {
        type: "string",
        required: false,
        description: "'HORIZONTAL', 'VERTICAL', or 'NONE' (default: 'NONE')",
      },
      padding: { type: "number", required: false, description: "Uniform padding (default: 0)" },
      item_spacing: {
        type: "number",
        required: false,
        description: "Gap between children (default: 0)",
      },
      fills: {
        type: "string",
        required: false,
        description: "Hex color for fill, e.g. '#FF0000'",
      },
    },
    returns: "{ nodeId, name, summary }",
  },
  async execute(params) {
    const frame = figma.createFrame();
    frame.name = params.name;
    frame.resize(params.width || 100, params.height || 100);
    frame.x = params.x || 0;
    frame.y = params.y || 0;

    const layout = params.auto_layout || "NONE";
    if (layout !== "NONE") {
      frame.layoutMode = layout as "HORIZONTAL" | "VERTICAL";
      frame.primaryAxisSizingMode = "AUTO";
      frame.counterAxisSizingMode = "AUTO";
    }

    if (params.padding) {
      const p = params.padding;
      frame.paddingTop = p;
      frame.paddingBottom = p;
      frame.paddingLeft = p;
      frame.paddingRight = p;
    }

    if (params.item_spacing) {
      frame.itemSpacing = params.item_spacing;
    }

    if (params.fills) {
      const hex = params.fills.replace("#", "");
      const r = parseInt(hex.substring(0, 2), 16) / 255;
      const g = parseInt(hex.substring(2, 4), 16) / 255;
      const b = parseInt(hex.substring(4, 6), 16) / 255;
      frame.fills = [{ type: "SOLID", color: { r, g, b } }];
    }

    figma.currentPage.appendChild(frame);
    return {
      nodeId: frame.id,
      name: frame.name,
      summary: `Created frame "${frame.name}" (${frame.width}x${frame.height}) at (${frame.x}, ${frame.y}).`,
    };
  },
});

// --- set_fills ---
registerOperation({
  manifest: {
    name: "set_fills",
    description: "Set the fill color of one or more nodes by ID.",
    category: "colors",
    params: {
      node_ids: {
        type: "string[]",
        required: true,
        description: "Array of node IDs to update",
      },
      color: {
        type: "string",
        required: true,
        description: "Hex color, e.g. '#3B82F6'",
      },
      opacity: {
        type: "number",
        required: false,
        description: "Fill opacity 0-1 (default: 1)",
      },
    },
    returns: "{ updated, failed, summary }",
  },
  async execute(params) {
    const hex = params.color.replace("#", "");
    const r = parseInt(hex.substring(0, 2), 16) / 255;
    const g = parseInt(hex.substring(2, 4), 16) / 255;
    const b = parseInt(hex.substring(4, 6), 16) / 255;
    const opacity = params.opacity !== undefined ? params.opacity : 1;

    let updated = 0;
    let failed = 0;
    for (const id of params.node_ids) {
      const node = figma.getNodeById(id);
      if (node && "fills" in node) {
        (node as GeometryMixin).fills = [{ type: "SOLID", color: { r, g, b }, opacity }];
        updated++;
      } else {
        failed++;
      }
    }
    figma.commitUndo();
    return {
      updated,
      failed,
      summary: `Set fill to ${params.color} on ${updated} nodes.${failed ? ` ${failed} nodes not found or not fillable.` : ""}`,
    };
  },
});

    name: "set_text",

        type: "string[]",
        required: true,
        description: "Array of text node IDs",

        type: "string",
        required: true,

      },

    let failed = 0;

        const textNode = node as TextNode;
        await figma.loadFontAsync(textNode.fontName as FontName);
        textNode.characters = params.text;
        updated++;
      } else {
        failed++;
      }
    }
    figma.commitUndo();
    return {
      updated,
      failed,
      summary: `Set text on ${updated} nodes.${failed ? ` ${failed} not found or not text.` : ""}`,
    };
  },
});
```

```typescript
registerOperation({
  manifest: {
    name: "move_node",
    description: "Move one or more nodes to a new position.",
    category: "layout",
    params: {
      node_ids: { type: "string[]", required: true, description: "Array of node IDs" },
      x: { type: "number", required: false, description: "New X position" },
      y: { type: "number", required: false, description: "New Y position" },
      dx: { type: "number", required: false, description: "Relative X offset" },
      dy: { type: "number", required: false, description: "Relative Y offset" },
    },
    returns: "{ moved, summary }",
  },
  async execute(params) {
    let moved = 0;
    for (const id of params.node_ids) {
      const node = figma.getNodeById(id);
      if (node && "x" in node) {
        const n = node as SceneNode;
        if (params.x !== undefined) n.x = params.x;
        if (params.y !== undefined) n.y = params.y;
        if (params.dx !== undefined) n.x += params.dx;
        if (params.dy !== undefined) n.y += params.dy;
        moved++;
      }
    }
    figma.commitUndo();
    return { moved, summary: `Moved ${moved} nodes.` };
  },

    description: "Resize one or more nodes to new dimensions.",

      node_ids: { type: "string[]", required: true, description: "Array of node IDs" },
      width: { type: "number", required: false, description: "New width" },
      height: { type: "number", required: false, description: "New height" },

        const h = params.height !== undefined ? params.height : n.height;

        resized++;
      }

registerOperation({
  manifest: {
    name: "delete_node",
    description: "Delete one or more nodes by ID.",
    category: "cleanup",
    params: {
      node_ids: { type: "string[]", required: true, description: "Array of node IDs to delete" },
    },
    returns: "{ deleted, not_found, summary }",
  },
  async execute(params) {
    let deleted = 0;
    let not_found = 0;
    for (const id of params.node_ids) {
      const node = figma.getNodeById(id);
      if (node && node.id !== figma.currentPage.id) {
        node.remove();
        deleted++;
      } else {
        not_found++;
      }
    }
    figma.commitUndo();
    return {
      deleted,
      not_found,
      summary: `Deleted ${deleted} nodes.${not_found ? ` ${not_found} not found.` : ""}`,
    };
  },
});

registerOperation({
  manifest: {
    name: "clone_node",
    description: "Clone a node and optionally reposition the copy.",
    category: "components",
    params: {
      node_id: { type: "string", required: true, description: "Node ID to clone" },
      x: { type: "number", required: false, description: "X position of clone" },
      y: { type: "number", required: false, description: "Y position of clone" },
      new_name: { type: "string", required: false, description: "Name for the clone" },
    },
    returns: "{ cloneId, cloneName, summary }",
  },
  async execute(params) {
    const node = figma.getNodeById(params.node_id);
    if (!node || !("clone" in node)) {
      return { cloneId: null, cloneName: null, summary: "Node not found or not cloneable." };
    }
    const clone = (node as SceneNode).clone();
    if (params.x !== undefined) clone.x = params.x;
    if (params.y !== undefined) clone.y = params.y;
    if (params.new_name) clone.name = params.new_name;
    return {
      cloneId: clone.id,
      cloneName: clone.name,
      summary: `Cloned "${node.name}" → "${clone.name}" (${clone.id}).`,
    };
  },
});
```

```typescript
// packages/bridge-plugin/src/operations/colors.ts
import { registerOperation } from "./registry";

function rgbToHex(r: number, g: number, b: number): string {
  const toHex = (c: number) =>
    Math.round(c * 255)
      .toString(16)
      .padStart(2, "0");
  return `#${toHex(r)}${toHex(g)}${toHex(b)}`;
}

// --- extract_palette ---
registerOperation({
  manifest: {
    name: "extract_palette",
    description:
      "Extract all unique solid fill colors used on the current page, with usage counts and which nodes use them.",
    category: "colors",
    params: {
      scope: { type: "string", required: false, description: "'page' (default) or 'selection'" },
    },
    returns: "{ colors: Array<{hex, count, nodeIds}>, total_unique, summary }",
  },
  async execute(params) {
    const scope = params.scope || "page";
    const nodes = scope === "selection" ? figma.currentPage.selection : figma.currentPage.findAll();

    const colorMap = new Map<string, { hex: string; count: number; nodeIds: string[] }>();

    for (const node of nodes) {
      if (!("fills" in node)) continue;
      const fills = (node as GeometryMixin).fills;
      if (!Array.isArray(fills)) continue;
      for (const fill of fills) {
        if (fill.type === "SOLID" && fill.visible !== false) {
          const hex = rgbToHex(fill.color.r, fill.color.g, fill.color.b);
          const entry = colorMap.get(hex) || { hex, count: 0, nodeIds: [] };
          entry.count++;
          if (entry.nodeIds.length < 10) entry.nodeIds.push(node.id);
          colorMap.set(hex, entry);
        }
      }
    }

    const colors = Array.from(colorMap.values()).sort((a, b) => b.count - a.count);
    return {
      colors: colors.slice(0, 200),
      total_unique: colors.length,
      summary: `Found ${colors.length} unique colors across ${nodes.length} nodes.`,
    };
  },
});

// --- find_non_style_colors ---
registerOperation({
  manifest: {
    name: "find_non_style_colors",
    description:
      "Find all nodes using hardcoded fill colors that are not linked to a local or library style.",
    category: "colors",
    params: {
      scope: { type: "string", required: false, description: "'page' (default) or 'selection'" },
    },
    returns: "{ violations: Array<{nodeId, nodeName, hex}>, count, summary }",
  },
  async execute(params) {
    const scope = params.scope || "page";
    const nodes = scope === "selection" ? figma.currentPage.selection : figma.currentPage.findAll();

    const violations: { nodeId: string; nodeName: string; hex: string }[] = [];

    for (const node of nodes) {
      if (!("fills" in node) || !("fillStyleId" in node)) continue;
      const styleId = (node as any).fillStyleId;
      if (styleId && styleId !== "" && styleId !== figma.mixed) continue;

      const fills = (node as GeometryMixin).fills;
      if (!Array.isArray(fills)) continue;
      for (const fill of fills) {
        if (fill.type === "SOLID" && fill.visible !== false) {
          violations.push({
            nodeId: node.id,
            nodeName: node.name,
            hex: rgbToHex(fill.color.r, fill.color.g, fill.color.b),
          });
          break;
        }
      }
    }

    return {
      violations: violations.slice(0, 200),
      count: violations.length,
      summary: `Found ${violations.length} nodes with hardcoded colors (no style linked).`,
    };
  },
});
```

```typescript
// packages/bridge-plugin/src/operations/typography.ts
import { registerOperation } from "./registry";

// --- audit_text_styles ---
registerOperation({
  manifest: {
    name: "audit_text_styles",
    description:
      "Audit all text nodes for style consistency. Reports font family, size, weight, and line height usage with counts.",
    category: "typography",
    params: {
      scope: { type: "string", required: false, description: "'page' (default) or 'selection'" },
    },
    returns:
      "{ styles: Array<{font, size, weight, lineHeight, count}>, total_text_nodes, unique_styles, summary }",
  },
  async execute(params) {
    const scope = params.scope || "page";
    const textNodes =
      scope === "selection"
        ? (figma.currentPage.selection.filter((n) => n.type === "TEXT") as TextNode[])
        : (figma.currentPage.findAll((n) => n.type === "TEXT") as TextNode[]);

    const styleMap = new Map<
      string,
      {
        font: string;
        size: number | string;
        weight: number | string;
        lineHeight: string;
        count: number;
      }
    >();

    for (const node of textNodes) {
      const fontName = node.fontName;
      const font = fontName === figma.mixed ? "mixed" : fontName.family;
      const size = node.fontSize === figma.mixed ? "mixed" : node.fontSize;
      const weight = fontName === figma.mixed ? "mixed" : fontName.style;
      const lh = node.lineHeight;
      let lineHeight = "auto";
      if (lh !== figma.mixed) {
        if (lh.unit === "AUTO") lineHeight = "auto";
        else if (lh.unit === "PIXELS") lineHeight = lh.value + "px";
        else if (lh.unit === "PERCENT") lineHeight = lh.value + "%";
      } else {
        lineHeight = "mixed";
      }

      const key = `${font}|${size}|${weight}|${lineHeight}`;
      const entry = styleMap.get(key) || { font, size, weight, lineHeight, count: 0 };
      entry.count++;
      styleMap.set(key, entry);
    }

    const styles = Array.from(styleMap.values()).sort((a, b) => b.count - a.count);
    return {
      styles: styles.slice(0, 100),
      total_text_nodes: textNodes.length,
      unique_styles: styles.length,
      summary: `${textNodes.length} text nodes using ${styles.length} unique style combinations.`,
    };
  },
});

// --- list_fonts ---
registerOperation({
  manifest: {
    name: "list_fonts",
    description: "List all fonts used in the file with usage counts.",
    category: "typography",
    params: {
      scope: { type: "string", required: false, description: "'page' (default) or 'selection'" },
    },
    returns: "{ fonts: Array<{family, styles, count}>, total_fonts, summary }",
  },
  async execute(params) {
    const scope = params.scope || "page";
    const textNodes =
      scope === "selection"
        ? (figma.currentPage.selection.filter((n) => n.type === "TEXT") as TextNode[])
        : (figma.currentPage.findAll((n) => n.type === "TEXT") as TextNode[]);

    const fontMap = new Map<string, { family: string; styles: Set<string>; count: number }>();

    for (const node of textNodes) {
      const fontName = node.fontName;
      if (fontName === figma.mixed) {
        const entry = fontMap.get("mixed") || {
          family: "mixed",
          styles: new Set<string>(),
          count: 0,
        };
        entry.count++;
        fontMap.set("mixed", entry);
        continue;
      }
      const entry = fontMap.get(fontName.family) || {
        family: fontName.family,
        styles: new Set<string>(),
        count: 0,
      };
      entry.styles.add(fontName.style);
      entry.count++;
      fontMap.set(fontName.family, entry);
    }

    const fonts = Array.from(fontMap.values())
      .map((f) => ({ family: f.family, styles: Array.from(f.styles), count: f.count }))
      .sort((a, b) => b.count - a.count);

    return {
      fonts,
      total_fonts: fonts.length,
      summary: `${fonts.length} font families across ${textNodes.length} text nodes.`,
    };
  },
});
```

```typescript
registerOperation({
  manifest: {
    name: "populate_text",
    description:
      "Fill text nodes with placeholder content (lorem ipsum) or custom text. Can target selection or all text nodes matching a name pattern.",
    category: "content",
    params: {
      text: {
        type: "string",
        required: false,
        description: "Custom text to populate. If omitted, uses lorem ipsum.",
      },
      name_pattern: {
        type: "string",
        required: false,
        description: "Regex pattern to match node names (e.g. 'placeholder|lorem')",
      },
      scope: { type: "string", required: false, description: "'page' (default) or 'selection'" },
      max_chars: {
        type: "number",
        required: false,
        description: "Max characters per node (default: no limit)",
      },
    },
    returns: "{ populated, summary }",
  },
  async execute(params) {
    const scope = params.scope || "page";
    const content = params.text || LOREM;
    const pattern = params.name_pattern ? new RegExp(params.name_pattern, "i") : null;

    const textNodes =
      scope === "selection"
        ? (figma.currentPage.selection.filter((n) => n.type === "TEXT") as TextNode[])
        : (figma.currentPage.findAll((n) => n.type === "TEXT") as TextNode[]);

    let populated = 0;
    for (const node of textNodes) {
      if (pattern && !pattern.test(node.name)) continue;
      await figma.loadFontAsync(node.fontName as FontName);
      const text = params.max_chars ? content.slice(0, params.max_chars) : content;
      node.characters = text;
      populated++;
    }

    figma.commitUndo();
    return {
      populated,
      summary: `Populated ${populated} text nodes.`,
    };
  },
});
```

```typescript
function rgbToHex(r: number, g: number, b: number): string {

    Math.round(c * 255)
      .toString(16)
      .padStart(2, "0");
  return `#${toHex(r)}${toHex(g)}${toHex(b)}`;
}

// --- extract_css ---
registerOperation({
  manifest: {
    name: "extract_css",
    description:
      "Extract CSS-like properties from selected nodes or the current page. Returns layout, typography, colors, and sizing as CSS key-value pairs.",
    category: "export",
    params: {
      scope: { type: "string", required: false, description: "'selection' (default) or 'page'" },
      max_nodes: {
        type: "number",
        required: false,
        description: "Max nodes to extract (default: 20)",
      },
    },
    returns: "{ nodes: Array<{nodeId, nodeName, css}>, count, summary }",
  },
  async execute(params) {
    const scope = params.scope || "selection";
    const maxNodes = params.max_nodes || 20;

    const nodes = scope === "selection" ? figma.currentPage.selection : figma.currentPage.findAll();

    const results: { nodeId: string; nodeName: string; css: Record<string, string> }[] = [];

    for (const node of nodes.slice(0, maxNodes)) {
      const css: Record<string, string> = {};

      // Dimensions
      if ("width" in node) {
        css["width"] = Math.round(node.width) + "px";
        css["height"] = Math.round((node as any).height) + "px";
      }

      // Position
      if ("x" in node) {
        css["left"] = Math.round(node.x) + "px";
        css["top"] = Math.round(node.y) + "px";
      }

      // Border radius
      if ("cornerRadius" in node) {
        const cr = (node as any).cornerRadius;
        if (cr !== figma.mixed && cr > 0) css["border-radius"] = cr + "px";
      }

      // Fills
      if ("fills" in node) {
        const fills = (node as any).fills;
        if (Array.isArray(fills) && fills.length > 0) {
          const fill = fills[0];
          if (fill.type === "SOLID" && fill.visible !== false) {
            const hex = rgbToHex(fill.color.r, fill.color.g, fill.color.b);
            const opacity = fill.opacity !== undefined ? fill.opacity : 1;
            css["background-color"] =
              opacity < 1
                ? `${hex}${Math.round(opacity * 255)
                    .toString(16)
                    .padStart(2, "0")}`
                : hex;
          }
        }
      }

      // Strokes
      if ("strokes" in node) {
        const strokes = (node as any).strokes;
        if (Array.isArray(strokes) && strokes.length > 0) {
          const stroke = strokes[0];
          if (stroke.type === "SOLID" && stroke.visible !== false) {
            const hex = rgbToHex(stroke.color.r, stroke.color.g, stroke.color.b);
            const weight = "strokeWeight" in node ? (node as any).strokeWeight : 1;
            css["border"] = `${weight}px solid ${hex}`;
          }
        }
      }

      // Text properties
      if (node.type === "TEXT") {
        const textNode = node as TextNode;
        const fontName = textNode.fontName;
        if (fontName !== figma.mixed) {
          css["font-family"] = fontName.family;
          css["font-weight"] = fontName.style.toLowerCase().includes("bold") ? "700" : "400";
        }
        const fontSize = textNode.fontSize;
        if (fontSize !== figma.mixed) css["font-size"] = fontSize + "px";
        const lh = textNode.lineHeight;
        if (lh !== figma.mixed && lh.unit === "PIXELS") css["line-height"] = lh.value + "px";
        else if (lh !== figma.mixed && lh.unit === "PERCENT") css["line-height"] = lh.value + "%";

        // Text color from fills
        const textFills = textNode.fills;
        if (Array.isArray(textFills) && textFills.length > 0 && textFills[0].type === "SOLID") {
          css["color"] = rgbToHex(textFills[0].color.r, textFills[0].color.g, textFills[0].color.b);
        }
      }

      // Auto-layout → flexbox
      if ("layoutMode" in node) {
        const n = node as FrameNode;
        if (n.layoutMode !== "NONE") {
          css["display"] = "flex";
          css["flex-direction"] = n.layoutMode === "HORIZONTAL" ? "row" : "column";
          css["gap"] = n.itemSpacing + "px";
          if (n.paddingTop > 0 || n.paddingBottom > 0 || n.paddingLeft > 0 || n.paddingRight > 0) {
            css["padding"] =
              `${n.paddingTop}px ${n.paddingRight}px ${n.paddingBottom}px ${n.paddingLeft}px`;
          }
        }
      }

      if (Object.keys(css).length > 0) {
        results.push({ nodeId: node.id, nodeName: node.name, css });
      }
    }

    return {
      nodes: results,
      count: results.length,
      summary: `Extracted CSS from ${results.length} nodes.`,
    };
  },
});
```

Rewrite `packages/mcp-server/src/websocket.ts`. Key changes:

- Add `activeFileKey: string | null` that auto-switches on interaction

```typescript
import { WebSocketServer, WebSocket } from "ws";
import type { IncomingMessage } from "http";
import { parseMessage, type ServerToPluginMessage, type ResultMessage } from "@pluginos/shared";

interface ConnectedFile {
  ws: WebSocket;
  fileKey: string;
  fileName: string;
  currentPage: string;
  lastActivity: number;
}

interface PendingRequest {
  resolve: (value: ResultMessage) => void;
  reject: (reason: Error) => void;
  timer: ReturnType<typeof setTimeout>;
}

interface WebSocketServerOptions {
  portRange?: [number, number];
}

export class PluginOSWebSocketServer {
  private wss: WebSocketServer | null = null;
  private files = new Map<string, ConnectedFile>();
  private activeFileKey: string | null = null;
  private port: number | null = null;
  private pending = new Map<string, PendingRequest>();
  private options: { portRange: [number, number] };

  constructor(options?: WebSocketServerOptions) {
    this.options = {
      portRange: options?.portRange ?? [9500, 9510],
    };
  }

  async start(): Promise<void> {
    const [min, max] = this.options.portRange;
    for (let port = min; port <= max; port++) {
      try {
        await this.tryPort(port);
        this.port = port;
        this.setupServer();
        return;
      } catch {
        continue;
      }
    }
    throw new Error(`No available port in range ${min}-${max}`);
  }

  private tryPort(port: number): Promise<void> {
    return new Promise((resolve, reject) => {
      const server = new WebSocketServer({
        port,
        verifyClient: (info: { origin: string; req: IncomingMessage }) => {
          const allowed = ["null", "https://www.figma.com", "https://figma.com"];
          return !info.origin || allowed.includes(info.origin);
        },
      });
      server.on("listening", () => {
        this.wss = server;
        resolve();
      });
      server.on("error", reject);
    });
  }

  private setupServer(): void {
    if (!this.wss) return;
    this.wss.on("connection", (ws: WebSocket) => {
      // We don't know the fileKey yet — it arrives via the first STATUS message.
      // Store a temporary reference until we get it.
      let fileKey: string | null = null;

      ws.on("message", (raw: Buffer) => {
        const msg = parseMessage(raw.toString());
        if (!msg) return;

        if (msg.type === "result") {
          const p = this.pending.get(msg.id);
          if (p) {
            clearTimeout(p.timer);
            this.pending.delete(msg.id);
            p.resolve(msg as ResultMessage);
          }
        } else if (msg.type === "status") {
          const key = msg.fileKey || "unknown";
          fileKey = key;
          this.files.set(key, {
            ws,
            fileKey: key,
            fileName: msg.fileName,
            currentPage: msg.currentPage,
            lastActivity: Date.now(),
          });
          this.activeFileKey = key;
        }
      });

      ws.on("close", () => {
        if (fileKey && this.files.has(fileKey)) {
          this.files.delete(fileKey);
          if (this.activeFileKey === fileKey) {
            // Switch active to most recent remaining file
            const remaining = Array.from(this.files.values());
            remaining.sort((a, b) => b.lastActivity - a.lastActivity);
            this.activeFileKey = remaining.length > 0 ? remaining[0].fileKey : null;
          }
        }
        // Reject pending requests targeting this client
        for (const [id, p] of this.pending) {
          clearTimeout(p.timer);
          this.pending.delete(id);
          p.reject(new Error("Plugin disconnected"));
        }
      });
    });
  }

  isConnected(fileKey?: string): boolean {
    if (fileKey) return this.files.has(fileKey);
    return this.files.size > 0;
  }

  getActiveFileKey(): string | null {
    return this.activeFileKey;
  }

  setActiveFile(fileKey: string): boolean {
    if (!this.files.has(fileKey)) return false;
    this.activeFileKey = fileKey;
    return true;
  }

  listFiles(): Array<{ fileKey: string; fileName: string; currentPage: string }> {
    return Array.from(this.files.values()).map((f) => ({
      fileKey: f.fileKey,
      fileName: f.fileName,
      currentPage: f.currentPage,
    }));
  }

  getStatus(): {
    connected: boolean;
    fileKey: string | null;
    fileName: string | null;
    currentPage: string | null;
    port: number | null;
    connectedFiles: number;
  } {
    const active = this.activeFileKey ? this.files.get(this.activeFileKey) : null;
    return {
      connected: this.files.size > 0,
      fileKey: active?.fileKey ?? null,
      fileName: active?.fileName ?? null,
      currentPage: active?.currentPage ?? null,
      port: this.port,
      connectedFiles: this.files.size,
    };
  }

  sendAndWait(
    message: ServerToPluginMessage,
    timeout = 30000,
    fileKey?: string
  ): Promise<ResultMessage> {
    return new Promise((resolve, reject) => {
      const targetKey = fileKey || this.activeFileKey;
      if (!targetKey || !this.files.has(targetKey)) {
        reject(
          new Error(
            fileKey
              ? `File "${fileKey}" not connected.`
              : "No plugin connected. Open PluginOS Bridge in Figma."
          )
        );
        return;
      }
      const file = this.files.get(targetKey)!;
      if (file.ws.readyState !== WebSocket.OPEN) {
        reject(new Error(`Connection to "${file.fileName}" is not open.`));
        return;
      }

      file.lastActivity = Date.now();
      this.activeFileKey = targetKey;

      const timer = setTimeout(() => {
        this.pending.delete(message.id);
        reject(new Error(`Operation timed out after ${timeout}ms`));
      }, timeout);

      this.pending.set(message.id, { resolve, reject, timer });
      file.ws.send(JSON.stringify(message));
    });
  }

  async close(): Promise<void> {
    for (const [id, p] of this.pending) {
      clearTimeout(p.timer);
      p.reject(new Error("Server closing"));
    }
    this.pending.clear();

    for (const f of this.files.values()) {
      f.ws.close();
    }
    this.files.clear();
    this.activeFileKey = null;

    return new Promise((resolve) => {
      if (this.wss) {
        this.wss.close(() => resolve());
      } else {
        resolve();
      }
    });
  }
}
```

```typescript
import { createServer, IncomingMessage, ServerResponse, Server } from "http";
import { readFileSync, existsSync } from "fs";
import { join, dirname } from "path";
import { fileURLToPath } from "url";

export function createHttpServer(getUiContent: () => string): Server {
  const server = createServer((req: IncomingMessage, res: ServerResponse) => {
    // CORS for Figma plugin iframe
    res.setHeader("Access-Control-Allow-Origin", "*");
    res.setHeader("Access-Control-Allow-Methods", "GET, OPTIONS");
    res.setHeader("Access-Control-Allow-Headers", "Content-Type");

    if (req.method === "OPTIONS") {
      res.writeHead(204);
      res.end();
      return;
    }

    if (req.url === "/ui" || req.url === "/ui.html") {
      res.writeHead(200, { "Content-Type": "text/html; charset=utf-8" });
      res.end(getUiContent());
      return;
    }

    if (req.url === "/health") {
      res.writeHead(200, { "Content-Type": "application/json" });
      res.end(JSON.stringify({ status: "ok" }));
      return;
    }

    res.writeHead(404);
    res.end("Not found");
  });

  return server;
}
```

```typescript
  ];
  for (const path of candidates) {
    if (existsSync(path)) {
      return readFileSync(path, "utf-8");
    }
  }
  return "<html><body><p>PluginOS UI not found. Run: npm run build -w packages/bridge-plugin</p></body></html>";
}

let cachedUi: string | null = null;

const httpServer = createHttpServer(() => {
  if (!cachedUi) cachedUi = loadUiContent();
  return cachedUi;
});

const wsServer = new PluginOSWebSocketServer({ httpServer });
await wsServer.start();
console.error(`PluginOS WebSocket + HTTP server on port ${wsServer.getStatus().port}`);

const mcpServer = createPluginOSServer(wsServer);
const transport = new StdioServerTransport();
await mcpServer.connect(transport);
console.error("PluginOS MCP server running on stdio");
```

- [ ] **Step 1: Create bootloader.html**

This is the minimal HTML that gets baked into the plugin via manifest. On load, it fetches the full UI from the MCP server's HTTP endpoint. If the server isn't running, it falls back to the bundled static version.

```html
<!-- packages/bridge-plugin/src/bootloader.html -->
<!DOCTYPE html>
<html>
  <head>
    <style>
      body {
        margin: 0;
        font-family: Inter, system-ui, sans-serif;
        font-size: 11px;
        background: var(--figma-color-bg);
        color: var(--figma-color-text);
        display: flex;
        align-items: center;
        padding: 8px 12px;
        gap: 8px;
      }
      .dot {
        width: 8px;
        height: 8px;
        border-radius: 50%;
        flex-shrink: 0;
      }
      .red {
        background: #f24822;
      }
      .green {
        background: #1bc47d;
      }
      .yellow {
        background: #f5a623;
      }
      #info {
        color: var(--figma-color-text-secondary, #999);
      }
    </style>
  </head>
  <body>
    <div class="dot red" id="dot"></div>
    <div>
      <div id="status">Loading...</div>
      <div id="info">Connecting to PluginOS server</div>
    </div>
    <script>
      // Bootloader: try to fetch full UI from MCP server, fall back to static bundled version
      var PORT_MIN = 9500;
      var PORT_MAX = 9510;

      function setStatus(connected, port, message) {
        var dot = document.getElementById("dot");
        var status = document.getElementById("status");
        var info = document.getElementById("info");
        dot.className = connected ? "dot green" : "dot red";
        status.textContent = connected ? "Connected (port " + port + ")" : "Disconnected";
        info.textContent =
          message || (connected ? "Ready for operations" : "Searching for server...");
      }

      async function tryBootload() {
        for (var port = PORT_MIN; port <= PORT_MAX; port++) {
          try {
            var res = await fetch("http://localhost:" + port + "/ui", {
              signal: AbortSignal.timeout(2000),
            });
            if (res.ok) {
              // Server found — inject the full UI
              var html = await res.text();
              document.open();
              document.write(html);
              document.close();
              return;
            }
          } catch (e) {
            // Try next port
          }
        }
        // No server found — use static bundled fallback
        setStatus(false, null, "Server not found. Run: npx pluginos");
        // Retry every 5 seconds
        setTimeout(tryBootload, 5000);
      }

      tryBootload();
    </script>
  </body>
</html>
```
